# Librarian interface
```Java
package my_first_project;

import java.util.ArrayList;

public interface Librarian {

    Book findBookByTitle(String title, ArrayList<Book> bookList);

    void printBookInfo(Book book);

    void addBook(Book book, ArrayList<Book> bookList);

    void removeBook(Book book, ArrayList<Book> bookList);

    void reviseBook(Book book, Book userBook);
}

```

# Librarianlmpl class
```Java
package my_first_project;

import java.util.ArrayList;

public class Librarianlmpl implements Librarian{

	@Override
	public Book findBookByTitle(String title, ArrayList<Book> bookList) {
		// TODO Auto-generated method stub
		
			if(bookList.contains(title)) {
				int number = bookList.indexOf(title);
				return bookList.get(number);
			}
			else {
				return null;
			}
		
	}

	@Override
	public void printBookInfo(Book book) {
		// TODO Auto-generated method stub
		System.out.println(book.getTitle());
		System.out.println(book.getAuthor());
		System.out.println(book.getDate());
	}

	@Override
	public void addBook(Book book, ArrayList<Book> bookList) {
		// TODO Auto-generated method stub
		bookList.add(book);
	}

	@Override
	public void removeBook(Book book, ArrayList<Book> bookList) {
		// TODO Auto-generated method stub
		bookList.remove(book);
	}

	@Override
	public void reviseBook(Book book, Book userBook) {
		// TODO Auto-generated method stub
		book.setTitle(userBook.getTitle());
		book.setAuthor(userBook.getAuthor());
		book.setDate(userBook.getDate());
	}

}
```

# Library class
```Java
package my_first_project;

import java.util.ArrayList;

public class Library {

    private ArrayList<Book> bookList = new ArrayList<>();

    private Book oldFuture = new Book();
    private Book dataStructure = new Book();
    private Book mind = new Book();
    private Book redAndBlack = new Book();

    public Library() {
        oldFuture.setTitle("나는 전설이다");
        oldFuture.setAuthor("몰라요");
        oldFuture.setDate("1991");
        bookList.add(oldFuture);

        dataStructure.setTitle("나무를 심은 사람");
        dataStructure.setAuthor("몰라요");
        dataStructure.setDate("2012");
        bookList.add(dataStructure);

        mind.setTitle("나무");
        mind.setAuthor("황석정");
        mind.setDate("1914");
        bookList.add(mind);

        redAndBlack.setTitle("변신,시골의사");
        redAndBlack.setAuthor("프랜치카프카");
        redAndBlack.setDate("1830");
        bookList.add(redAndBlack);
    }

    public ArrayList<Book> getBookList() {
        return bookList;
    }
}
```

# User class
```Java
package my_first_project;

import java.util.Scanner;

public class User {

    Scanner sc = new Scanner(System.in);

    public boolean isEntered(){
        return sc.hasNext();
    }

    public Book createBook() {
        Book book = new Book();
        book.setTitle(sc.next());
        book.setAuthor(sc.next());
        book.setDate(sc.next());
        return book;
    }

    public String input() {
        return sc.nextLine();
    }

    public String inputTitle() {
        return sc.nextLine();
    }

}

```

# Main class
```Java
package my_first_project;

public class MainTest {

	public static void main(String[] args) {
		// TODO Auto-generated method stub
			Library library = new Library(); 
			Librarianlmpl librarianlmpl = new Librarianlmpl();
			User user = new User();
			
			switch(user.input()) {
			case "제목으로 찾기":
				if(librarianlmpl.findBookByTitle(user.inputTitle(), library.getBookList())!=null) {
				librarianlmpl.printBookInfo(librarianlmpl.findBookByTitle(user.inputTitle(), library.getBookList())); 
				System.out.println("출력되었습니다");
				}
				else {
					System.out.println("일치하는 책이 없습니다");
				}
				break;
			case "책 추가":
				librarianlmpl.addBook(user.createBook(),library.getBookList());
				System.out.println("추가되었습니다");
				break;
			case "책 제거":
				if(librarianlmpl.findBookByTitle(user.inputTitle(), library.getBookList())!=null) {
				librarianlmpl.removeBook(librarianlmpl.findBookByTitle(user.inputTitle(), library.getBookList()), library.getBookList());
				System.out.println("제거되었습니다");
				}
				else {
					System.out.println("제거할 책이 존재하지 않습니다");
				}
				break;
			case "책 수정":
				if(librarianlmpl.findBookByTitle(user.inputTitle(), library.getBookList())!=null) {
				librarianlmpl.reviseBook(librarianlmpl.findBookByTitle(user.inputTitle(), library.getBookList()), user.createBook());
				System.out.println("수정되었습니다");
				}
				else {
					System.out.println("수정할 책이 존재하지 않습니다");
				}
				break;
			case "종료":
				System.out.print("시스템이 종료되었습니다");
				break;
			}
	}

}

```

## 코드
- 맨 위에 package는 옛날 실습 파일에다가 쓰다보니 default값이 아닌 패키지명이 생겨서 있는것임
- Library 클래스를 받을 때 글씨가 깨져서 나와서 해당 안에 있는 내용은 임의로 수정함

## 리뷰
- 사실상 건드린 파일은 Main과 인터페이스를 참조 받은 Librarianlmpl 파일 밖에 없고 나머지 파일은 객체를 만들어서 다 public으로 되어있으니 가져다가 쓴 작업 밖에 안했음
- 내부 코드를 짜는 것에 어려움보다는 객체가 여러 개여서 필요한 부분만 맞춰서 직관적으로 쓰면서 그리고 ArrayList 같은 경우도 구글링을 참조하여서 해당 메소드를 사용해서 씀
