- 리사이클러뷰에 활용한 어댑터
```Java
public class PlateAdapter extends RecyclerView.Adapter<PlateAdapter.PlateViewHolder> { // 리사이클러뷰에 대해서 어댑터 생성후 Adapter 참조함 

    private List<PlateModel> plateModelList; // PlateModel은 생성자로써 별도 클래스를 만들어 놓음 이를 통해서 사용할 이미지에 대해서 데이터를 그대로 받아올 수 있게됨
    private Context context;


    public PlateAdapter(List<PlateModel> plateModelList, Context context) {
        this.plateModelList = plateModelList;
        this.context = context;
    }

    @NonNull
    @Override // 아이템 뷰를 위한 뷰홀더 객체 리턴하여서 생성하고 리턴함
    public PlateViewHolder onCreateViewHolder(@NonNull ViewGroup viewGroup, int viewType) {

        View view = LayoutInflater.from(viewGroup.getContext()).inflate(R.layout.layout_plates,viewGroup,false); // layout_plates에 정의된 Resource를 View 객체로 전환함, 이를 통해서 아이템 뷰를 위한 뷰홀더 객체 리턴하여서 생성하는 과정을 할 수 있음
        return new PlateViewHolder(view);
    }
    // position에 해당하는 데이터를 뷰홀더의 아이템뷰에 표시함
    @Override
    public void onBindViewHolder(@NonNull PlateViewHolder holder, int position) {

        PlateModel plateModel = plateModelList.get(position);

        Glide.with(context).load(plateModel.getPlate_img()).into(holder.plateImg); // glide 라이브러리, ViewHolder 안에 이미지를 넣는 방법

    }
    // 전체 데이터 갯수를 리턴함
    @Override
    public int getItemCount() {
        return plateModelList.size();
    }
    // 아이템 뷰를 저장하는 뷰홀더 클래스, 연습앱에서 이미지를 저장하기 위해서 사용함, 생성자를 통해서 받은 이미지 데이터를 받아서 이미지를 보일 수 있게함
    public class PlateViewHolder extends RecyclerView.ViewHolder {

        private ImageView plateImg;

        public PlateViewHolder(@NonNull View itemView) {
            super(itemView);

            plateImg = (ImageView) itemView.findViewById(R.id.imageView2);
        }
    }
}
```
