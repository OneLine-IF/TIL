#define _CRT_SECURE_NO_WARNINGS
#include<stdio.h>
#include<stdlib.h>
#include<time.h>// to revise time seed
#include<windows.h>// to measure the time of process 

void insertion_sort(int* arr, int n) {// insertion sort method
	int tmp, j;

	for (int k = 1; k < n; k++) {
		tmp = arr[k];
		j = k - 1;
		while (arr[j] > tmp && j >= 0) {
			arr[j + 1] = arr[j];
			j--;
		}
		arr[j + 1] = tmp;
	}
}

void insertion_sort_reverse(int* arr, int n) {// insertion sort for make reverse order of the array
	int tmp, j;

	for (int k = 1; k < n; k++) {
		tmp = arr[k];
		j = k - 1;
		while (arr[j] < tmp && j >= 0) {
			arr[j + 1] = arr[j];
			j--;
		}
		arr[j + 1] = tmp;
	}
}

int Biggest(int* arr, int k) {// to find the most biggest number and use it in a selection_sort
	int max = arr[k], pos = k;

	for (int j = k - 1; j >= 0; j--) {
		if (max < arr[j]) {
			max = arr[j];
			pos = j;
		}
	}
	return pos;
}

void selection_sort(int* arr, int n) {// selection sort method
	int pos, tmp;

	for (int k = n - 1; k > 0; k--) {
		pos = Biggest(arr, k);
		tmp = arr[k];
		arr[k] = arr[pos];
		arr[pos] = tmp;
	}
}

void main() {
	srand(time(NULL));

	LARGE_INTEGER ticksPerSec;
	LARGE_INTEGER start, end, diff;

	int n, * arr1 = NULL, * arr2 = NULL;

	printf("\n>> Case A <<\n\n");// change the name of the case (A or B or C)
	printf("---------------------------------------------------------------------------------\n");
	for (int i = 0; i < 5; i++) {// 5 times repeat
		printf("Size: ");
		scanf("%d", &n);
		printf("\n");
		arr1 = (int*)malloc(n * sizeof(int));// use this in selection sort
		arr2 = (int*)malloc(n * sizeof(int));// use this in insertion sort

		for (int i = 0; i < n; i++) {
			*(arr1 + i) = rand() % 10001;
			*(arr2 + i) = *(arr1 + i);
		}

		//insertion_sort(arr1, n); // Case_B: insertion_sort   Case_C: insertion_sort_reverse
		//insertion_sort(arr2, n); // Case_B: insertion_sort   Case_C: insertion_sort_reverse

		// selection sort
		QueryPerformanceFrequency(&ticksPerSec);
		QueryPerformanceCounter(&start);
		selection_sort(arr1, n);
		QueryPerformanceCounter(&end);
		diff.QuadPart = end.QuadPart - start.QuadPart;
		printf("selection_Sort: %.12f sec\n", ((double)diff.QuadPart / (double)ticksPerSec.QuadPart));

		// insertion sort
		QueryPerformanceFrequency(&ticksPerSec);
		QueryPerformanceCounter(&start);
		insertion_sort(arr2, n);
		QueryPerformanceCounter(&end);
		diff.QuadPart = end.QuadPart - start.QuadPart;
		printf("Insertion_Sort: %.12f sec\n", ((double)diff.QuadPart / (double)ticksPerSec.QuadPart));
		
		printf("---------------------------------------------------------------------------------\n");
	}
}
