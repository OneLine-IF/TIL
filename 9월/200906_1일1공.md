# 자료구조 中 트리 파트를 복습하였음

- 아래의 문제: pre, post, in, level order를 + 트리의 원하는 위치에 리프 추가하기 

```C
#define _CRT_SECURE_NO_WARNINGS
#include <stdio.h>
#include <string.h>
#include <stdlib.h>
struct Node {
	struct Node *llink, *rlink, *plink;
	int e;
};
struct Que {
	struct Que *prev, *next;
	struct Node *data;
};
struct Node *getNode(struct Node **p) {
	(*p) = (struct Node *)malloc(sizeof(struct Node));
	(*p)->llink = NULL;
	(*p)->rlink = NULL;
	(*p)->plink = NULL;
	return (*p);
}
struct Que *getQue(struct Que **p) {
	(*p) = (struct Que *)malloc(sizeof(struct Que));
	(*p)->prev = NULL;
	(*p)->next = NULL;
	return (*p);
}
void enqueue(struct Que **front, struct Que **rear, struct Node *data) {
	struct Que *p = getQue(&p);
	p->data = data;

	if (*rear == NULL) {
		*front = p;
		*rear = p;
	}
	else {
		(*rear)->next = p;
		p->prev = (*rear);
		(*rear) = p;
	}
}
struct Node *dequeue(struct Que **front, struct Que **rear) {
	struct Que *p = *front;
	if ((*front)->next == NULL) {
		*front = NULL;
		*rear = NULL;
	}
	else {
		(*front) = (*front)->next;
		(*front)->prev = NULL;
	}
	return p->data;
}
struct Node *findParent(struct Node *p, int e) {
	if (p == NULL) return NULL;
	else if (p->e == e) return p;

	struct Node *left = findParent(p->llink, e);
	struct Node *right = findParent(p->rlink, e);
	if (left == NULL) return right;
	else return left;
}
void edge(struct Node *r, int x, int y, int z) {
	struct Node *lnode = getNode(&lnode), *rnode = getNode(&rnode), *p = findParent(r, x);
	lnode->e = y; rnode->e = z;

	if (p == NULL) { r->e = x; p = r; }
	
	if (y != 0) { p->llink = lnode; lnode->plink = p; }
	if (z != 0) { p->rlink = rnode; rnode->plink = p; }
}
void preOrder(struct Node *p) {
	if (p != NULL) {
		printf(" %d", p->e);
		preOrder(p->llink);
		preOrder(p->rlink);
	}
}
void postOrder(struct Node *p) {
	if (p != NULL) {
		postOrder(p->llink);
		postOrder(p->rlink);
		printf(" %d", p->e);
	}
}
void inOrder(struct Node *p) {
	if (p != NULL) {
		inOrder(p->llink);
		printf(" %d", p->e);
		inOrder(p->rlink);
	}
}
void levelOrder(struct Node *v) {
	struct Que *front = NULL, *rear = NULL;
	
	enqueue(&front, &rear, v);
	while (front != NULL) {
		v = dequeue(&front, &rear);
		printf(" %d", v->e);
		if (v->llink != NULL) enqueue(&front, &rear, v->llink);
		if (v->rlink != NULL) enqueue(&front, &rear, v->rlink);
	}
}

	int N, x, y, z;
	scanf("%d", &N);
	for (int i = 0; i < N; i++) {
		scanf("%d%d%d", &x, &y, &z);
		edge(root, x, y, z);
	}
	preOrder(root); printf("\n");
	inOrder(root); printf("\n");
	postOrder(root); printf("\n");
	levelOrder(root); printf("\n");
}
```
