#define _CRT_SECURE_NO_WARNINGS
#include<stdio.h>
#include<stdlib.h>
#include<time.h>
#include<windows.h>

void insertion_sort(int* arr, int n) {
	int tmp, j;

	for (int k = 1; k < n; k++) {
		tmp = arr[k];
		j = k - 1;
		while (arr[j] > tmp && j >= 0) {
			arr[j + 1] = arr[j];
			j--;
		}
		arr[j + 1] = tmp;
	}
}

int Biggest(int* arr, int k) {
	int max = arr[k], pos = k;

	for (int j = k - 1; j >= 0; j--) {
		if (max < arr[j]) {
			max = arr[j];
			pos = j;
		}
	}

	return pos;
}

void selection_sort(int* arr, int n) {
	int pos, tmp;

	for (int k = n - 1; k > 0; k--) {
		pos = Biggest(arr, k);
		tmp = arr[k];
		arr[k] = arr[pos];
		arr[pos] = tmp;
	}
}

void main() {
	srand(time(NULL));

	LARGE_INTEGER ticksPerSec;
	LARGE_INTEGER start, end, diff;

	int n, * arr1 = NULL, * arr2 = NULL;
	scanf("%d", &n);
	arr1 = (int*)malloc(n * sizeof(int));
	arr2 = (int*)malloc(n * sizeof(int));

	for (int i = 0; i < n; i++) {
		*(arr1 + i) = rand() % 10001;
		*(arr2 + i) = *(arr1 + i);
	}

	for (int i = 0; i < n; i++) {
		printf(" %d", *(arr1 + i));
	}
	printf("\n");
	for (int i = 0; i < n; i++) {
		printf(" %d", *(arr2 + i));
	}
	printf("\n");

	// 선택정렬
	QueryPerformanceFrequency(&ticksPerSec);
	QueryPerformanceCounter(&start);
	selection_sort(arr1, n); // 실행시간 측정을 하고 하는 것
	QueryPerformanceCounter(&end);
	// 측정값으로부터 실행시간 계산
	diff.QuadPart = end.QuadPart - start.QuadPart;
	printf("selection_Sort: %.12f sec\n", ((double)diff.QuadPart / (double)ticksPerSec.QuadPart));

	// 삽입정렬
	QueryPerformanceFrequency(&ticksPerSec);
	QueryPerformanceCounter(&start);
	insertion_sort(arr2, n); // 실행시간 측정을 하고 하는 것
	QueryPerformanceCounter(&end);
	// 측정값으로부터 실행시간 계산
	diff.QuadPart = end.QuadPart - start.QuadPart;
	printf("Insertion_Sort: %.12f sec\n", ((double)diff.QuadPart / (double)ticksPerSec.QuadPart));

}
